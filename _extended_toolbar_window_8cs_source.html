59  static void EnsureWindowExistsUpdate()
60  {
61  foreach (var instance in toolbarInstances)
62  {
63  if(instance.window == null)
64  {
65  var windows = Resources.FindObjectsOfTypeAll(instance.type);
66 
67  if(windows.Length > 1)
68  {
69  Debug.LogError("Found more than one instance of " + instance.type);
70  }
71  else if (windows.Length == 1)
72  {
73  instance.window = (ExtendedToolbarWindow) windows[0];
74  }
75 
76  if(instance.window == null)
77  {
78  instance.window = (ExtendedToolbarWindow) CreateInstance(instance.type);
79  instance.window.position = Rect.zero;
80  instance.window.ShowPopup();
81  }
82 
83  instance.window.horizontalOffset = instance.horizontalOffset;
84  instance.window.width = instance.width;
85  }
86  }

161  void UpdatePosition()
162  {

166  var window = windowProperty.GetValue(toolbarRef, new object[0]);
167 
168  var positionProperty = GetProperty(window.GetType(), "position");
169  var windowPosition = (Rect) positionProperty.GetValue(window, null);
170 
171  var rect = new Rect();
